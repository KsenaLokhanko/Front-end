function changePix(nameImg, newR){
    var img = new SimpleImage (nameImg);
    for (var pixel of img.values()){
        if (pixel.getRed() <14){
        pixel.setRed(newR);}
    }
    return img;
}

var img = new SimpleImage("smallhands.png");
print("jgjgj", img);
var res = changePix(img, 125);
print(res);
